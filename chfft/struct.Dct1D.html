Perform a discrete cosine transform

Example

extern crate chfft;

use chfft::Dct1D;

fn main() {
    let input = [2.0, 0.0, 1.0, 1.0, 0.0, 3.0, 2.0, 4.0];

    let mut dct = Dct1D::<f64>::new(2, input.len(), false);
    let output = dct.forward(&input);
    println!("the transform of {:?} is {:?}", input, output);
}
